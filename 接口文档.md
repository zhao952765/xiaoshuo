### 2.2 核心接口

```typescript
// AI 响应/请求
interface AIResponse<T = any> { success: boolean; data?: T; error?: string; }
interface AIRequest { prompt: string; model?: string; temperature?: number; }

// 小说项目（含感情线/肉欲线关联）
interface Novel {
  id: string; title: string; summary: string; adultMode: boolean;
  tags: string[]; targetWords: NovelLength;
  characters: string[]; worldSettings: string[]; chapters: string[]; plotLines: string[];
  emotionArcId: string | null; lustArcId: string | null;
  createdAt: number; updatedAt: number;
}

// 角色（含 NSFW 扩展）
interface Character {
  id: string; name: string; roleType: RoleType; avatar: string;
  basicInfo: { age: string; gender: string; occupation: string; };
  appearance: string; personality: string[]; background: string; abilities: string;
  relationships: Array<{ targetId: string; targetName: string; type: string; description: string; }>;
  voice: string; innerWorld: string; arc: string; tags: string[];
  nsfwProfile?: { bodyType: string; sensitiveZones: string[]; sexualTraits: string[]; fetishTags: string[]; experienceLevel: string; };
  createdAt: number; updatedAt: number;
}

// 世界观
interface WorldSetting {
  id: string; name: string; worldType: WorldType; description: string; overview: string;
  rules: Array<{ name: string; description: string; scope: string; limit: string; sideEffect: string; }>;
  locations: Array<{ name: string; type: string; description: string; atmosphere: string; scenes: string[]; }>;
  timeline: Array<{ era: string; title: string; description: string; impact: string; }>;
  society: string; culture: string; economy: string;
  createdAt: number; updatedAt: number;
}

// 章节
interface Chapter {
  id: string; title: string; summary: string; content: string;
  order: number; status: ChapterStatus; volumeId: string | null;
  wordCount: number; mood: string; characters: string[]; hooks: string; tags: string[];
  createdAt: number; updatedAt: number;
}

// 感情线（React Flow）
interface EmotionArc {
  id: string; novelId: string;
  nodes: EmotionArcNode[];   // id/type(position/type/data{label/description/intensity/characterIds/chapterId/color})
  edges: EmotionArcEdge[];   // id/source/target/type/label/animated/style
  timeline: EmotionArcEvent[];
  createdAt: number; updatedAt: number;
}

// 肉欲线
interface LustArc {
  id: string; novelId: string;
  intensityCurve: LustIntensityPoint[];  // id/chapterId/chapterTitle/value/description/characters/sceneType/order
  climaxPoints: LustClimaxPoint[];       // id/chapterId/chapterTitle/intensity/description/characters/type(tease/buildup/climax/afterglow)/order
  createdAt: number; updatedAt: number;
}

// 标签 / 记忆 / AI 模型 / 日志 / 卷 / 对话
interface Tag { id: string; name: string; category: TagCategory; color: string; isFavorite: boolean; }
interface Memory { id: string; type: MemoryType; content: string; source: string; tags: string[]; modelName: string | null; projectId: string | null; timestamp: number; duration: number | null; }
interface AIModel { id: string; name: string; apiKey: string; baseUrl: string; modelType: AIModelType; modelId: string; temperature: number; maxTokens: number; stream: boolean; isDefault: boolean; }
interface Log { id: string; type: LogType; message: string; detail: string | null; timestamp: number; }
interface Volume { id: string; novelId: string; name: string; summary: string; order: number; chapters: string[]; statusColor: string; }
interface Conversation { id: string; title: string; messages: ChatMessage[]; context: { characters: boolean; worldSettings: boolean; chapters: boolean; }; }

// 一键推导结果
interface OneClickResult {
  title: string; summary: string;
  protagonist: Character; supporting: Character[];
  worldSetting: WorldSetting; plotLine: PlotLine;
  chapters: Array<{ title: string; summary: string; }>;
  firstChapter: string;
  emotionArc: EmotionArc; lustArc: LustArc; tags: Tag[]; prompts: any[];
}
```

## 5. 数据流

```
用户操作
  → 页面组件调用 Store Action（useStore / useAppStore）
    → Zustand set() 更新状态
      → persist 中间件自动同步到 localStorage（键名: private-novel-studio-pro-storage）
        → autoSave 定时器（每 autoSaveInterval 秒）→ 全量写入
        → beforeunload / visibilitychange 事件 → saveNow() 强制写入
```

**自动备份**：`autoBackup=true` 时，每轮保存自动复制到 `backup-{timestamp}`，保留 10 个。

**数据恢复**：
```typescript
saveNow()                    // 手动触发保存
listBackups()                // 列出备份
restoreFromBackup(key)       // 从备份恢复
```

**版本迁移**：
- v0 → v1：从旧键 `private-novel-studio-pro-storage` 恢复数据
- v1 → v2：新增 `emotionEvents`/`volumes` 字段

## 6. 工具函数 API

### 6.1 promptLoader.ts

**文件**: `src/renderer/utils/promptLoader.ts`

| 函数 | 签名 | 说明 |
|------|------|------|
| `loadPrompt(filename)` | `string → string` | 加载 `.md` 提示词文件（Vite glob 编译时导入） |
| `loadPromptWithVars(filename, vars)` | `(string, Record<string,string>) → string` | 加载 + 替换 `{key}` 占位符 |
| `listPrompts()` | `→ string[]` | 列出所有可用提示词文件名 |

提示词文件位于项目根目录 `prompts/*.md`。

### 6.2 markdownParser.ts

**文件**: `src/renderer/utils/markdownParser.ts`

| 函数 | 说明 |
|------|------|
| `parseMarkdownFields(text)` | 3 策略提取字段：1) 先按 `**字段**` 分块解析 2) 按行扫描匹配 3) 纯段落后备提取姓名/年龄/性别/外貌 |
| `parseProtagonistFields(text)` | 主角设定智能解析，开头纯段落自动归为外貌 |
| `cleanMarkdown(text)` | 去除粗体 `**`、列表标记 `-•·`、数字序号、压缩空行 |
| `extractNameFromText(text)` | 3 级兜底：`**姓名**：XXX` → `姓名：XXX` → 首行 2-6 字词 |

### 6.3 deduceParser.ts

**文件**: `src/renderer/utils/deduceParser.ts`

| 函数 | 说明 |
|------|------|
| `isTitleLine(line)` | 严格模式标题行判断（排除姓名/性别/年龄等非标题短词） |
| `parseGenerationResult(text)` | 两遍扫描法：1) 定位字段标题 2) 按区间提取内容。支持后备：独立第一章检测、感情线宽泛匹配、大纲关键词、"第X章"收集 |
| `extractProtagonistName(text)` | 3 级兜底：键值对 → 标题行后缀 → 首行 2-4 字中文名 |
| `parseSupportingChars(text, now, gender?)` | 解析配角列表，支持 3 种分割策略（`**名字**` / 数字序号 / 空行），预过滤章节标题和无意义条目 |
| `parseWorldview(text)` | 解析世界观全文：概述/规则/地点/时间线/社会/文化/经济 |
| `extractSubSection(text, titles)` | 子区块提取（兼容多种标题格式） |

### 6.4 characterAI.ts
